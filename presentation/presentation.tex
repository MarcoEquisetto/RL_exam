\section{Methodology}
%---------------------------------------------------------

\begin{frame}{Introduction to World Models}
    \begin{itemize}
        \item \textbf{The Problem:} Traditional Model-Free RL (e.g., PPO, SAC) requires millions of interactions with the real environment.
        \item \textbf{The Solution:} World Models learn a compressed simulator of the environment. The agent can then train entirely in "imagination."
        \item \textbf{Benefits:}
        \begin{itemize}
            \item High sample efficiency.
            \item Safe exploration (the agent fails in its imagination, not in reality).
            \item Learning from offline data.
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{DreamerV4 Architecture}
    \begin{itemize}
        \item Unlike previous versions (DreamerV1-V3) which used Recurrent State Space Models (RSSM), \textbf{DreamerV4} uses a highly scalable \textbf{Block-Causal Transformer}.
        \item \textbf{Key Components:}
        \begin{itemize}
            \item \textbf{Tokenizer (Encoder \& Decoder):} Uses Masked Autoencoders (MAE) to compress image patches into robust discrete latents.
            \item \textbf{Dynamics Model:} A causal transformer that predicts the next latent state given past states and actions.
            \item \textbf{Actor-Critic:} The agent policy that learns from the Dynamics Model's imagined rollouts.
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Shortcut Forcing Objective}
    \begin{itemize}
        \item A major innovation in DreamerV4 is \textbf{Shortcut Forcing}.
        \item Instead of predicting $t+1, t+2, t+3$ strictly sequentially, the model learns to directly denoise the state at step $t+k$.
        \item This allows the model to "jump" into the future during imagination, dramatically speeding up the generation of long-horizon behaviors.
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Code Deep-Dive: The Dynamics Model}
    The Dynamics Model predicts future latent states $z_{t+1}$ conditioned on continuous actions.
    \begin{minted}[fontsize=\footnotesize, bgcolor=lightgray]{python}
class Dynamics(nn.Module):
    def forward(self, actions, step_idxs, signal_idxs, packed_enc_tokens):
        # Embed current spatial state
        spatial_tokens = self.spatial_proj(packed_enc_tokens)
        
        # Embed continuous actions
        action_tokens = self.action_encoder(actions)
        
        # Concatenate tokens and apply Transformer
        tokens = torch.cat([action_tokens, spatial_tokens, ...], dim=2)
        x = self.transformer(tokens)
        
        # Output predicted next spatial state
        spatial_out = x[:, :, self.spatial_slice, :]
        x1_hat = self.flow_x_head(spatial_out)
        return x1_hat
    \end{minted}
\end{frame}

\section{Conclusion}

\begin{frame}{Conclusion}
    \begin{itemize}
        \item \textbf{Scalability:} DreamerV4 proves that Transformer-based world models scale beautifully and efficiently.
        \item \textbf{Offline Learning:} Capable of learning purely from offline datasets without risky environment interactions.
        \item \textbf{Implementation Feasibility:} While full training requires significant compute (8x A100s for days), running inference and fine-tuning on continuous control benchmarks is highly accessible.
